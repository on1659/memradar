function sigmoid(x: number, midpoint: number, steepness: number): number {
  if (!Number.isFinite(x)) return 0.5
  return 1 / (1 + Math.exp(-steepness * (x - midpoint)))
}

function extractProject(cwd: string): string {
  const parts = cwd.replace(/\\/g, '/').split('/').filter(Boolean)
  // ※ 실제 구현: Math.min(parts.length, 6) — Windows/Unix 경로를 모두 커버하기 위해 4 → 6 상향
  return parts.slice(0, Math.min(parts.length, 6)).join('/')
}

function median(arr: number[]): number {
  if (arr.length === 0) return 0
  const sorted = [...arr].sort((a, b) => a - b)
  const mid = Math.floor(sorted.length / 2)
  return sorted.length % 2 ? sorted[mid] : (sorted[mid - 1] + sorted[mid]) / 2
}

rhythm: { label: ['스프린터', '마라토너'], value: rhythmValue } — value 높을수록 마라토너(M)

※ 실제 구현에서는 label 배열이 [낮은값, 높은값] 규약을 지키도록 style/rhythm을 반전 배치. TypeCode 매핑은 계획 그대로 유지.

매핑 규칙 (TypeCode 문자열 불변)

styleValue ≥ 0.5 → 'E' (설계자)  |  < 0.5 → 'R' (탐험가)
scopeValue ≥ 0.5 → 'W' (유목민)  |  < 0.5 → 'D' (한우물)
rhythmValue ≥ 0.5 → 'M' (마라토너)  |  < 0.5 → 'S' (스프린터)
